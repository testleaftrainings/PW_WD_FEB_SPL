1. String Immutability & Memory
Question: Strings are immutable. What does "modifying" a string actually mean?
Answer: Immutability means the data inside a string cannot be changed once created.
The Reality: When you "modify" a string, JavaScript creates a new string in a new memory location.
The Reference: The variable name simply stops pointing to the old memory address and starts pointing to the new one.

let str = "test"; // Points to memory location A
str = str + "leaf"; // Creates "testleaf" in location B; str now points to B

2. Extracting Data with split()
Task: Extract 12345 from 'Order ID: #12345-Confirmed'.
Solution: Use split() to break the string at specific delimiters.

let productDetails = "Order ID: #12345-Confirmed";
let prod_status = productDetails.split("#"); // ["Order ID: ", "12345-Confirmed"]
let prod_id = prod_status[1].split('-');    // ["12345", "Confirmed"]

console.log(prod_id[0]); // Output: 12345

3. Handling "Falsy" Values (The 0 Bug)
Scenario: if (data) treats 0 as false, failing validation for quantity fields.
Solution: Check specifically for null and undefined using strict inequality.

function isValid(data) {
    // Explicitly allow 0 by only blocking null or undefined
    return data !== null && data !== undefined && data !== "";
}

console.log(isValid(0)); // Returns: true

4. Correcting Characters in a String
Question: Does str[0] = 'J' work?
Answer: No. Because strings are immutable, bracket notation is read-only. To change a character, you must reconstruct the string.

let str = 'Hello World';
str[0] = 'J'; // Silently fails; str is still 'Hello World'

// Correct Approach:
str = 'J' + str.slice(1); 
console.log(str); // 'Jello World'


5. Parsing Server Logs
Task: Get an array of timestamps only where an ERROR occurred.
Solution: Combine split() with match() and Regular Expressions.

const logs = "ERROR: 404 [Time: 12:00] | INFO: Login Success [Time: 12:01] | ERROR: 500 [Time: 12:05]";
const entries = logs.split("|");

for (const entry of entries) {
    if (entry.includes("ERROR")) {
        const time = entry.match(/\[Time: (.*?)\]/);
        console.log(time[1]); // Output: 12:00, 12:05
    }
}


6. Logic: Finding the Longest Word
Task: Identify the longest word in a sentence.
Solution: Loop through an array of words and track the maximum length found.

let str = 'Automation testing is incredibly important';
let words = str.split(" ");
let longest = "";

for (let i = 0; i < words.length; i++) {
    if (words[i].length > longest.length) {
        longest = words[i];
    }
}
console.log(longest); // "incredibly"


7. replaceAll() in Test Automation
Use Case: Essential for Data Sanitization.
Automation scripts often scrape prices or IDs that contain symbols ($, ,). Use replaceAll() with a regex to clean the data for mathematical comparisons.

const expectedPrice = "$1,000.50";
// Removes everything except numbers and the decimal point
const cleanPrice = expectedPrice.replaceAll(/[^0-9.]/g, ""); 
console.log(cleanPrice); // "1000.50"


8. Removing Duplicates
Task: Remove duplicate characters from a string.
Solution: Use a loop to check if a character already exists in a "results" array.

let word = "programming";
let letters = word.split("");
let result = [];

for (let i = 0; i < letters.length; i++) {
    if (!result.includes(letters[i])) {
        result.push(letters[i]);
    }
}
console.log(result.join("")); // "progamin"
